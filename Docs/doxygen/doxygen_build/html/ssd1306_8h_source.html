19/* Control byte for i2c
20Co : bit 8 : Continuation Bit
21 * 1 = no-continuation (only one byte to follow)
22 * 0 = the controller should expect a stream of bytes.
23D/C# : bit 7 : Data/Command Select bit
24 * 1 = the next byte or byte stream will be Data.
25 * 0 = a Command byte or byte stream will be coming up next.
26 Bits 6-0 will be all zeros.
27Usage:
280x80 : Single Command byte
290x00 : Command Stream
300xC0 : Single Data byte
310x40 : Data Stream
32*/
33#define OLED_CONTROL_BYTE_CMD_SINGLE 0x80
34#define OLED_CONTROL_BYTE_CMD_STREAM 0x00
35#define OLED_CONTROL_BYTE_DATA_SINGLE 0xC0
36#define OLED_CONTROL_BYTE_DATA_STREAM 0x40
